public ListNode reverseList(ListNode head) {
    // 哨兵
    ListNode dummy = new ListNode(0);
    // 当前节点
    ListNode node = head;
    // 下一个节点
    ListNode nextNode;
    while(node != null){
        // 保存下一个节点的
        nextNode = node.next;
        // 插入节点到dummy节点后
        node.next = dummy.next;
        dummy.next = node;
        // 更新当前节点
        node = nextNode;
    }
    return dummy.next;
}
