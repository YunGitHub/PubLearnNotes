```java
public boolean isPalindrome(int x) {
    // 负数不是回文数
    if (x < 0) {
        return false;
    }
    // 0 为回文
    if (x == 0) {
        return true;
    }
    // 个位数为 0 不是回文数
    if (x % 10 == 0) {
        return false;
    }
    int m = x;
    int n = 0;
    while (m != 0) {
        n = n * 10 + m % 10;
        m = m / 10;
    }
    return n == x;
}
```
