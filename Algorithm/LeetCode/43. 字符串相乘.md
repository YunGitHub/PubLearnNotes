public String multiply(String num1, String num2) {
    if (num1.equals("0") || num2.equals("0")) {
        return "0";
    }
    String result = "0";
    int s1 = num1.length();
    int s2 = num2.length();
    // 拆解为 n2 轮乘法
    StringBuffer zero = new StringBuffer();
    for (int n2 = s2 - 1; n2 >= 0; n2--) {
        // 从二轮开始都会进行错位一个0
        if (n2 != s2 - 1) {
            zero.append("0");
        }
        int v2 = num2.charAt(n2) - '0';
        // 乘法
        StringBuffer curr = new StringBuffer();
        int carry = 0;
        for (int n1 = s1 - 1; n1 >= 0; n1--) {
            int v1 = num1.charAt(n1) - '0';
            int product = v1 * v2 + carry;
            curr.append(product % 10);
            carry = product / 10;
        }
        if (carry != 0) {
            curr.append(carry);
        }
        // 每一轮乘法之后进行错位相加
        result = addStrings(result, curr.reverse().append(zero).toString());
    }
    return result;
}

// 字符串相加
public String addStrings(String num1, String num2) {
    int n1 = num1.length() - 1;
    int n2 = num2.length() - 1;
    int carry = 0;
    StringBuffer result = new StringBuffer();
    while (n1 >= 0 || n2 >= 0 || carry != 0) {
        int v1 = n1 >= 0 ? num1.charAt(n1) - '0' : 0;
        int v2 = n2 >= 0 ? num2.charAt(n2) - '0' : 0;
        int sum = v1 + v2 + carry;
        result.append(sum % 10);
        carry = sum / 10;
        n1 --;
        n2 --;
    }
    // 计算完以后的答案需要翻转过来
    result.reverse();
    return result.toString();
}
